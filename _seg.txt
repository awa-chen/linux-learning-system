function openDoc(key) {
    var md = (window.__DOCS__ || {})[key];
    if (md == null) return;
    var reader = document.getElementById("reader");
    if (reader) { reader.classList.remove("open"); reader.innerHTML = ""; }
    var secs = document.querySelectorAll("#progress-sec, #roadmap-sec, #sandbox-sec");
    Array.prototype.forEach.call(secs, function (s) { s.hidden = true; });
    var docSec = document.getElementById("doc-sec");
    docSec.hidden = false;
    document.getElementById("doc-title").textContent = DOC_TITLES[key] || key;
    document.getElementById("doc-body").innerHTML = mdToHtml(md);
    document.getElementById("doc-back").onclick = closeDoc;
    docSec.scrollIntoView({ behavior: "smooth" });
    if (location.hash !== "#doc=" + key) history.replaceState(null, "", "#doc=" + key);
  }
  function closeDoc() {
    var docSec = document.getElementById("doc-sec");
    docSec.hidden = true;
    var secs = document.querySelectorAll("#progress-sec, #roadmap-sec, #sandbox-sec");
    Array.prototype.forEach.call(secs, function (s) { s.hidden = false; });
    if (location.hash.indexOf("#doc=") === 0) history.replaceState(null, "", "#");
  }
